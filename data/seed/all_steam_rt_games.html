Bazzite Linux — RT Quick Reference (RX 9070 XT + Mesa 26.0 RADV)

Verificar setup

# Verificar que RADV esta activo vulkaninfo | grep driverName # Debe decir: radv # Verificar version Mesa vulkaninfo | grep driverVersion # Forzar RADV si AMDVLK interfiere AMD_VULKAN_ICD=RADV %command%

FSR4 Redstone (RDNA4 nativo)

# Activar FSR4 en cualquier juego DX12 via Proton PROTON_FSR4_UPGRADE=1 SteamDeck=0 %command% # Con indicador visual de FSR4 PROTON_FSR4_UPGRADE=1 PROTON_FSR4_INDICATOR=1 SteamDeck=0 %command%

Gamescope + FSR (compositor)

# Renderizar a 1800p, salida 4K con FSR gamescope -h 1800 -H 2160 -F fsr -f -- %command% # Con MangoHud gamescope -h 1800 -H 2160 -F fsr -f -- MANGOHUD=1 %command%

Variables RT utiles

# Template para juegos RT problematicos RADV_DEBUG=nodcc %command% # Fix VRAM crashes con reBAR/SAM VKD3D_CONFIG=no_upload_hvv %command% # Combo problematico completo AMD_VULKAN_ICD=RADV RADV_DEBUG=nodcc VKD3D_CONFIG=no_upload_hvv %command% # Emular RT por software (GPUs sin HW RT) RADV_PERFTEST=emulate_rt %command%

Capcom RE Engine (RE2, RE3, RE4, REV, RE Requiem, MHW)

# Desbloquear RT en juegos Capcom que detectan Wine %command% /WineDetectionEnabled:False
