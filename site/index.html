# Verify RADV is active vulkaninfo | grep driverName # Must say: radv # Check Mesa version vulkaninfo | grep driverVersion # Force RADV if AMDVLK interferes AMD_VULKAN_ICD=RADV %command%

FSR4 Redstone (RDNA4)

# Enable FSR4 on any DX12 game via Proton PROTON_FSR4_UPGRADE=1 SteamDeck=0 %command% # With FSR4 visual indicator PROTON_FSR4_UPGRADE=1 PROTON_FSR4_INDICATOR=1 SteamDeck=0 %command%

Gamescope + FSR

# Render at 1800p, output 4K with FSR gamescope -h 1800 -H 2160 -F fsr -f -- %command% # With MangoHud gamescope -h 1800 -H 2160 -F fsr -f -- MANGOHUD=1 %command%

# Template for problematic RT games RADV_DEBUG=nodcc %command% # Fix VRAM crashes with reBAR/SAM VKD3D_CONFIG=no_upload_hvv %command% # Full problematic combo AMD_VULKAN_ICD=RADV RADV_DEBUG=nodcc VKD3D_CONFIG=no_upload_hvv %command% # Emulate RT in software (GPUs without HW RT) RADV_PERFTEST=emulate_rt %command%
